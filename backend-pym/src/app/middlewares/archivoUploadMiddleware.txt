function getSafeExtension(mimetype) {
  return mimeToExt[mimetype] || '.bin';
}

const storage = multer.diskStorage({
  destination: (req, file, cb) => {
    let folder = 'documentos'; // Folder por defecto
    
    // 1. Prioridad: folder específico por fieldname
    const validation = fieldValidations[file.fieldname];
    if (validation && validation.folder) {
      folder = validation.folder;
    }
    
    // 2. Prioridad: folder desde body (si se especifica)
    if (req.body.folder && req.body.folder.trim() !== '') {
      folder = req.body.folder;
    }
    
    
    
    const destFolder = path.join(__dirname, '..', 'public', folder);
    fs.ensureDirSync(destFolder);
    cb(null, destFolder);
  },
  filename: (req, file, cb) => {
    // Sanitizar nombre y forzar extensión segura
    const nameWithoutExt = path.basename(file.originalname, path.extname(file.originalname));
    const sanitizedName = nameWithoutExt.replace(/[^a-zA-Z0-9-_]/g, '_');
    const safeExtension = getSafeExtension(file.mimetype);
    const filename = `${Date.now()}-${sanitizedName}${safeExtension}`;
    cb(null, filename);
  }
});

// Función auxiliar para validación de archivos - CORREGIDA
function validateFile(file, validation, req, cb) { // ← AÑADIDO req como parámetro
  // Validar MIME type
  if (!validation.mimes.includes(file.mimetype)) {
    const allowedTypes = validation.mimes.map(mime => {
      if (mime.startsWith('image/')) return mime.split('/')[1].toUpperCase();
      if (mime.startsWith('application/')) {
        if (mime.includes('pdf')) return 'PDF';
        if (mime.includes('msword')) return 'DOC';
        if (mime.includes('openxmlformats')) return 'DOCX';
      }
      return mime;
    }).join(', ');
    
    return cb(new Error(`Tipo de archivo no permitido para ${validation.description}. Tipos válidos: ${allowedTypes}`), false);
  }

  // Validar tamaño (usando content-length como aproximación)
  if (req && req.headers['content-length'] && parseInt(req.headers['content-length']) > validation.maxSize) {
    return cb(new Error(`Archivo demasiado grande para ${validation.description}. Tamaño máximo: ${validation.maxSize / 1024 / 1024}MB`), false);
  }

  cb(null, true);
}

const upload = multer({
  storage: storage,
  limits: {
    fileSize: 1024 * 1024 * 5 // Límite global de 5MB
  },
  fileFilter: (req, file, cb) => {
    const validation = fieldValidations[file.fieldname];
    
    // Si no hay validación específica, usar validación por defecto según el tipo
    if (!validation) {
      const defaultValidation = file.fieldname.includes('documento') 
        ? fieldValidations.documento 
        : fieldValidations.imagen;
      
      return validateFile(file, defaultValidation, req, cb); // ← PASA req
    }
    
    validateFile(file, validation, req, cb); // ← PASA req
  }
});

// Manejo de errores mejorado
upload.errorHandler = (err, req, res, next) => {
  if (err instanceof multer.MulterError) {
    switch (err.code) {
      case 'LIMIT_FILE_SIZE':
        return res.status(400).json({ 
          success: false,
          error: 'El archivo es demasiado grande. Tamaño máximo: 5MB' 
        });
      case 'LIMIT_UNEXPECTED_FILE':
        return res.status(400).json({ 
          success: false,
          error: 'Campo de archivo no permitido o inesperado' 
        });
      case 'LIMIT_PART_COUNT':
        return res.status(400).json({ 
          success: false,
          error: 'Demasiadas partes en el formulario' 
        });
      case 'LIMIT_FIELD_KEY':
        return res.status(400).json({ 
          success: false,
          error: 'Nombre de campo demasiado largo' 
        });
      default:
        return res.status(400).json({ 
          success: false,
          error: `Error al subir archivo: ${err.message}` 
        });
    }
  } else if (err) {
    return res.status(400).json({ 
      success: false,
      error: err.message 
    });
  }
  next();
};

// Exportar también las validaciones por si se necesitan externamente
